95template <typename TCode, typename TDesc>
96struct TResult {
97
98 Error::Kind kind;
99 std::optional<TCode> code;
100 std::optional<TDesc> desc;
101
103 constexpr bool ok() const { return kind == Error::Kind::Ok; }
105 constexpr operator bool() const { return ok(); }
106
108 friend std::ostream& operator << (std::ostream& out, const TResult& result) {
109 out << "[" << kind_as_string(result.kind) << "]";
110 if( !result.ok() ) {
111 out
112 << " Code=" << result.code.value_or(Error::Code<TCode>::Unspecified)
113 << " Desc=\"" << (result.desc.has_value() ? result.desc.value() : "") << "\"";
114 }
115 return out;
116 }
117
119 std::string as_string() {
120 std::ostringstream buf;
121 buf << *this;
122 return buf.str();
123 }
124
