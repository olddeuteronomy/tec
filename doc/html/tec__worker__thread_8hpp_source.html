64template <typename Traits>
65class WorkerThread: public Worker<Traits> {
66public:
67 using id_t = std::thread::id;

144 Lock lk(mtx_status_);
145 status_ = status;

148 // Sets the signal on exiting from `thread_proc'.
149 struct OnExit {
150 Signal& sig_;
151
152 OnExit(Signal& sig): sig_{sig} {}
153 ~OnExit() { sig_.set(); }
154 };
155
156 //-----------------------------------------------------------------
157 // WE HIDE THREAD PROCEDURE INSIDE A STRUCT TO INSTANTIATE
158 // DIFFERENT STATIC PROCEDURES OWNED BY DIFFERENT WORKERS.
159 //-----------------------------------------------------------------
160 template <typename traits>
161 struct details {
162 static void thread_proc(WorkerThread<traits>& wt) {
163 TEC_ENTER("WorkerThread::thread_proc");
164
165 // `sig_terminated' will signal on exit.
166 OnExit on_terminating{wt.sig_terminated_};
167
168 // We obtain thread_id and then we are waiting for `sig_begin'
169 // to resume the thread, see run().
170 wt.thread_id_ = std::this_thread::get_id();
171 TEC_TRACE("thread {} created.", wt.id());
172 wt.sig_running_.wait();
173 TEC_TRACE("`sig_running' received.");
174
175 // Initialize the Worker and set the result.
176 TEC_TRACE("on_init() called ...");
177 auto init_status = wt.on_init();
178 TEC_TRACE("on_init() returned {}.", init_status);
179 wt.set_status(init_status);
180
181 // Signals the Worker is inited, no matter if initialization failed.
182 wt.sig_inited_.set();
183 TEC_TRACE("`sig_inited' signalled.");
184
185 if( wt.status() ) {
186 // Start message polling if inited successfully.
187 TEC_TRACE("entering message loop.");
188 bool stop = false;
189 do {
190 auto msg = wt.mq_.dequeue();
191 TEC_TRACE("received Message [cmd={}].", msg.command);
192 if( !msg.quit() ) {
193 wt.dispatch(msg);
194 }
195 else {
196 stop = true;
197 }
198 } while( !stop );
199 TEC_TRACE("leaving message loop.");
200 }
201
202 // Finalize the Worker ONLY IF it has been inited successfully.
203 if( wt.status() ) {
204 TEC_TRACE("on_exit() called ...");
205 auto exit_status = wt.on_exit();
206 TEC_TRACE("on_exit() returned {}.", exit_status);
207 wt.set_status(exit_status);
208 }
209
210 // `sig_terminated' signals on exit, see OnExit struct defined above.
211 }
212 };

214public:
215
228 bool send(const Message& msg) override {
229 TEC_ENTER("WorkerThread::send");
230 if( status() && thread_.joinable() ) {
231 mq_.enqueue(msg);
232 TEC_TRACE("Message [cmd={}] sent.", msg.command);
233 return true;
234 }
235 else {
236 // No message loop.
237 return false;
238 }
239 }
240
250 Status run() override {
251 Lock lk{mtx_thread_proc_};
252 TEC_ENTER("WorkerThread::run");
253
254 if( !thread_.joinable() ) {
255 // No thread exists, possible system failure.
256 TEC_TRACE("no active thread.");
257 return {"no active thread", Error::Kind::System};
258 }
259
260 if( flag_running_ ) {
261 TEC_TRACE("WARNING: Worker thread is already running!");
262 return {};
263 }
264
265 // Resume the thread.
266 flag_running_ = true;
267 sig_running_.set();
268 TEC_TRACE("`sig_running' signalled.");
269
270 // Wait for thread initialization completed.
271 TEC_TRACE("waiting for `sig_inited' signalled ...");
272 sig_inited_.wait();
273
274 // Return the result of thread initialization.
275 return status();
276 }
277
289 Status terminate() override {
290 Lock lk{mtx_thread_proc_};
291 TEC_ENTER("WorkerThread::terminate");
292
293 if( flag_terminated_ ) {
294 TEC_TRACE("WARNING: Worker thread already terminated!");
295 return status();
296 }
297
298 if( !thread_.joinable() ) {
299 TEC_TRACE("WARNING: no thread exists!");
300 return status();
301 }
302
303 // Send Message::QUIT if normal exiting.
304 if( status() ) {
305 send(quit<Message>());
306 TEC_TRACE("QUIT sent.");
307 }
308
309 // Waits for the thread to finish its execution
310 TEC_TRACE("waiting for thread {} to finish ...", id());
311 thread_.join();
312 TEC_TRACE("thread {} finished OK.", id());
313
314 flag_terminated_ = true;
315 return status();
316 }
