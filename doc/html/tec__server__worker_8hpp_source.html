118 Status on_init() override {
119 TEC_ENTER("ServerWorker::on_init");
120
121 // Resume the server thread.
122 sig_run_server_thread_.set();
123
124 // Wait for the server to start.
125 if (!sig_started_.wait_for(this->params().start_timeout)) {
126 // Timeout!
127 TEC_TRACE("!!! Error: server start timeout -- server detached");
128 server_thread_->detach();
129 return {"Server start timeout", Error::Kind::TimeoutErr};
130 }
131 if (!status_started_) {
132 // Something went wrong; join the server thread.
133 server_thread_->join();
134 return status_started_;
135 }
136
137 // Everything is OK.
138 return {};
139 }
140
148 Status on_exit() override {
149 TEC_ENTER("ServerWorker::on_exit");
150
151 if (!server_thread_->joinable()) {
152 // Already finished or not started.
153 return {};
154 }
155
156 // Start a thread to shut down the server.
157 std::thread shutdown_thread([&] {
158 server_->shutdown(sig_stopped_);
159 });
160
161 // Wait for the server to shut down.
162 if (!sig_stopped_.wait_for(this->params().shutdown_timeout)) {
163 // Timeout! Detach the server thread.
164 server_thread_->detach();
165 TEC_TRACE("!!! Error: Server shutdown timeout.");
166 status_stopped_ = {"Server shutdown timeout", Error::Kind::TimeoutErr};
167 } else {
168 server_thread_->join();
169 }
170
171 shutdown_thread.join();
172 return status_stopped_;
173 }
