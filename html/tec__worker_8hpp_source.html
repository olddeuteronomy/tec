233 template <typename Derived, typename T>
234 void register_callback(Derived* worker, void (Derived::*callback)(const Message& msg)) {
235 Lock lk{mtx_slots_};
236 TEC_ENTER("Worker::register_callback");
237 //
238 // Ensure Derived is actually derived from Worker<Params>.
239 //
240 static_assert(std::is_base_of_v<Worker<Params>, Derived>,
241 "Derived must inherit from tec::Worker");
242 std::type_index ndx = std::type_index(typeid(T));
243 //
244 // Remove existing handler.
245 //
246 if (auto slot = slots_.find(ndx); slot != slots_.end()) {
247 slots_.erase(ndx);
248 }
249 //
250 // Set the slot.
251 //
252 slots_[ndx] = std::make_unique<Slot>(
253 worker,
254 [callback](Worker<Params>* worker, const Message& msg) {
255 // Safely downcast to Derived.
256 auto derived = dynamic_cast<Derived*>(worker);
257 (derived->*callback)(msg);
258 });
259 TEC_TRACE("Callback {} registered.", typeid(T).name());
260 }
261
262protected:
270 virtual void dispatch(const Message& msg) {
271 auto ndx = std::type_index(msg.type());
272 if (auto slot = slots_.find(ndx); slot != slots_.end()) {
273 slot->second->callback(slot->second->worker, std::cref(msg));
274 }
275 }

286 template <typename Params>
287 struct details {
295 static void thread_proc(Worker<Params>& wt) {
296 TEC_ENTER("Worker::thread_proc");
297 // Signal termination on exit.
298 Signal::OnExit on_terminating{&wt.sig_terminated_};
299 //
300 // Obtain thread ID and wait for the running signal.
301 //
302 wt.thread_id_ = std::this_thread::get_id();
303 TEC_TRACE("thread {} created.", wt.id());
304 wt.sig_running_.wait();
305 TEC_TRACE("`sig_running' received.");
306 //
307 // Exit immediately if flagged.
308 //
309 if (wt.flag_exited_) {
310 return;
311 }
312 //
313 // Initialize the worker.
314 //
315 TEC_TRACE("on_init() called ...");
316 wt.status_ = wt.on_init();
317 TEC_TRACE("on_init() returned {}.", wt.status_);
318 //
319 // Signal initialization complete.
320 //
321 wt.sig_inited_.set();
322 TEC_TRACE("`sig_inited' signalled.");
323
324 if (wt.status_) {
325 //
326 // Process messages if initialized successfully.
327 //
328 TEC_TRACE("entering message loop.");
329 bool stop = false;
330 do {
331 auto msg = wt.mq_.dequeue();
332 TEC_TRACE("received Message [{}].", name(msg));
333 if (is_null(msg)) {
334 stop = true;
335 wt.flag_exited_ = true;
336 } else {
337 wt.dispatch(std::cref(msg));
338 }
339 } while (!stop);
340 TEC_TRACE("leaving message loop, {} message(s) left in queue...", wt.mq_.size());
341 }
342 //
343 // Finalize if it was initialized successfully.
344 //
345 if (wt.status_) {
346 TEC_TRACE("on_exit() called ...");
347 wt.status_ = wt.on_exit();
348 TEC_TRACE("on_exit() returned {}.", wt.status_);
349 }
350 }
351 };

377 Status run() override {
378 Lock lk{mtx_thread_proc_};
379 TEC_ENTER("Worker::run");
380 //
381 // Create a new thread.
382 //
383 if (!thread_.joinable()) {
384 thread_ = std::thread(details<Params>::thread_proc, std::ref(*this));
385 }
386 else {
387 TEC_TRACE("`Worker::thread_proc' is already running.");
388 return {};
389 }
390 //
391 // Resume the thread.
392 //
393 flag_running_ = true;
394 sig_running_.set();
395 TEC_TRACE("`sig_running' signalled.");
396 //
397 // Wait for thread initialization completed.
398 //
399 TEC_TRACE("waiting for `sig_inited' signalled ...");
400 sig_inited_.wait();
401 // Return a result of thread initialization.
402 return status_;
403 }
404
413 Status terminate() override {
414 Lock lk{mtx_thread_proc_};
415 TEC_ENTER("Worker::terminate");
416 if (!thread_.joinable()) {
417 TEC_TRACE("WARNING: no thread exists!");
418 return status_;
419 }
420 if (!flag_running_) {
421 TEC_TRACE("Exiting the suspended thread...");
422 flag_exited_ = true;
423 sig_running_.set();
424 }
425 //
426 // Send the null message on normal exiting.
427 //
428 if (status_ && !flag_exited_) {
429 send(nullmsg());
430 TEC_TRACE("QUIT sent.");
431 }
432 //
433 // Wait for the thread to finish its execution.
434 //
435 TEC_TRACE("waiting for thread {} to finish ...", id());
436 thread_.join();
437 TEC_TRACE("thread {} finished OK.", id());
438
439 return status_;
440 }
